# Design Tokens

<Description>
  这是我们项目的设计令牌系统。它定义了颜色、字体、间距等基础视觉元素，是构建一致用户体验的基石。

## 颜色 (Colors)

export const ColorPalette = () => {{
  const [{{ globals }}] = useGlobals();
  const colors = globals.tokens.color;

  const renderColorSwatches = (colorGroup) => (
    <div style={{ display: 'flex', flexWrap: 'wrap', gap: '1rem', marginBottom: '2rem' }}>
      {Object.entries(colorGroup).map(([name, token]) => (
        typeof token.value === 'string' && (
          <div key={{name}} style={{ display: 'flex', flexDirection: 'column', alignItems: 'center' }}>
            <div style={{ width: '100px', height: '100px', backgroundColor: token.value, border: '1px solid #ddd', borderRadius: '8px' }} />
            <div style={{ marginTop: '0.5rem', textAlign: 'center' }}>
              <div style={{ fontWeight: 'bold' }}>{name}</div>
              <div style={{ fontSize: '0.8rem', color: '#666' }}>{token.value}</div>
              <div style={{ fontSize: '0.8rem', color: '#666' }}>{token.description}</div>
            </div>
          </div>
        )
      ))}
    </div>
  );

  return (
    <div>
      <h3>Primitive Colors</h3>
      {renderColorSwatches(colors.gray)}
      <h3>Brand Colors</h3>
      {renderColorSwatches(colors.brand)}
      <h3>Semantic Colors</h3>
      {renderColorSwatches(colors.semantic.bg)}
      {renderColorSwatches(colors.semantic.text)}
    </div>
  );
}};
