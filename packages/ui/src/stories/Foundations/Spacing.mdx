# Spacing

The spacing scale is used for margins, paddings, and gaps between elements. It is based on a 4px grid.

export const SpacingBlock = ({ name, size }) => (
  <div style={{ display: 'flex', alignItems: 'center', marginBottom: '1rem' }}>
    <div style={{ height: size, width: size, backgroundColor: '#0057ff', marginRight: '1rem', flexShrink: 0 }}></div>
    <div style={{ fontFamily: 'sans-serif' }}>
      <div style={{ fontWeight: 'bold', color: '#333' }}>{name}</div>
      <div style={{ color: '#666' }}>{size}</div>
    </div>
  </div>
);

<div style={{ display: 'flex', flexDirection: 'column' }}>
  {Object.entries(tokens.space).map(([name, value]) => (
    <SpacingBlock key={name} name={`space.${name}`} size={value} />
  ))}
</div>
