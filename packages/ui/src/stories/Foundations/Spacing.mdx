# Spacing (间距)

间距系统是构建一致、和谐布局的基础。我们使用基于 4px 倍数的间距令牌。

export const SpacingBlock = ({ space, value }) => (
  <div style={{ display: 'flex', alignItems: 'center', marginBottom: '10px' }}>
    <div style={{ width: '120px', fontFamily: 'monospace' }}>{space}</div>
    <div style={{ width: '60px', color: '#666' }}>{value}</div>
    <div style={{ height: value, width: value, backgroundColor: 'var(--accent-primary)' }} />
  </div>
);
